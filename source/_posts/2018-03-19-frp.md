### 在VPS上配置frps.ini
```
[common]
bind_port = 7777
#vhost_http_port网站访问端口
vhost_http_port = 80
#vhost_https_port = 443
#dashboard_port状态以及代理统计信息展示,网址:7500可查看详情
dashboard_port = 6443
log_file = ./frps.log
log_level = info
log_max_days = 3
#privilege_mode 特权模式,开通后web,ssh等使用都可以直接在客户端设置
privilege_mode = true
#特权连接密码
privilege_token = 12345qwert
#max_pool_count最大链接池,每个代理预先与后端服务器建立起指定数量的最大链接数
max_pool_count = 50
```

### 在NAS上配置frpc.ini
```
[common]
server_addr = 换成frps所在的服务器IP
#要与frps.ini中的bind_port一样
server_port = 7777
log_file = ./frpc.log
log_level = info
log_max_days = 3
#特权连接密码,要与frps.ini的一样
privilege_token = 12345qwert

[ssh]
privilege_mode = true
type = tcp
local_ip = 127.0.0.1
local_port = 22
remote_port = 6022
use_gzip = true
use_encryption = true
pool_count = 2
#访问方式 ssh -oPort=6022 test@x.x.x.x
#访问方式 ssh -p 6022 test@x.x.x.x

[nas]
privilege_mode = true
type = http
local_ip = 127.0.0.1
local_port = 5000
#custom_domains域名
custom_domains = nas.xrr.fun
pool_count = 10
#访问方式 http://nas.xrr.fun

[ws]
privilege_mode = true
type = http
local_ip = 127.0.0.1
local_port = 8888
custom_domains = ws.xrr.fun
pool_count = 10
#访问方式 http://ws.xrr.fun
```
